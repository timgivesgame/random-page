<!DOCTYPE html> <html lang="en"> <head> <meta charset="UTF-8"> <meta name="viewport" content="width=device-width, initial-scale=1.0"> <title>Random Animated Webpage</title> <style> body { margin: 0; overflow: hidden; background-color: #333; font-family: monospace; color: #ffffff; } .circle { position: absolute; border-radius: 50%; z-index: 1; } </style> </head> <body> <script> const numCircles = 20; const circles = []; for (let i = 0; i < numCircles; i++) { const circle = document.createElement('div'); circle.classList.add('circle'); circle.style.width = circle.style.height = `${Math.random() * 50 + 10}px`; circle.style.backgroundColor = `hsl(${Math.random() * 360}, 100%, 50%)`; circle.style.top = `${Math.random() * window.innerHeight}px`; circle.style.left = `${Math.random() * window.innerWidth}px`; circles.push({ element: circle, speedX: Math.random() * 4 - 2, speedY: Math.random() * 4 - 2 }); document.body.appendChild(circle); } function animate() { circles.forEach(circle => { let x = parseFloat(circle.element.style.left); let y = parseFloat(circle.element.style.top); x += circle.speedX; y += circle.speedY; if (x <= 0 || x >= window.innerWidth - parseFloat(circle.element.style.width)) { circle.speedX *= -1; } if (y <= 0 || y >= window.innerHeight - parseFloat(circle.element.style.height)) { circle.speedY *= -1; } circle.element.style.left = `${x}px`; circle.element.style.top = `${y}px`; }); requestAnimationFrame(animate); } animate(); </script> </body> </html>
